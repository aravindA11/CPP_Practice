 Exercise 2.3: What output will the following code produce?
 unsigned u = 10, u2 = 42;
 std::cout << u2 - u << std::endl; Ans: 32
 std::cout << u - u2 << std::endl; Ans: 4294967264
 int i = 10, i2 = 42;
 std::cout << i2 - i << std::endl; Ans: 32
 std::cout << i - i2 << std::endl; Ans: -32
 std::cout << i - u << std::endl; Ans: 0
 std::cout << u - i << std::endl; Ans: 0
